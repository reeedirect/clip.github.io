#   请在此添加实现代码   #
# ********** Begin *********#

v=sqrt(m*g/u)*tanh(sqrt(u*g/m)*t)
x=m/u*log(cosh(sqrt(u*g/m)*t))
print(f'当t={t}秒时，速度v={v:.2f}米/秒')
print(f'{t}秒后，小球位置为向下{x:.2f}米')
